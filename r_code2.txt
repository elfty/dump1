mydata = read.csv("mydata.csv", sep = " ")
plot(sample(mydata))
mydata$RATIO <- mydata$SHUCK / mydata$WHOLE
ggplot(mydata, aes(x = SEX, y = RATIO)) + geom_boxplot()
ggplot(mydata, aes(x = CLASS, y = RATIO)) + geom_boxplot()
mydata$VOLUME <- mydata$LENGTH * mydata$DIAM * mydata$HEIGHT
ggplot(mydata, aes(x = SEX, y = VOLUME)) + geom_boxplot()
ggplot(mydata, aes(x = CLASS, y = VOLUME)) + geom_boxplot()
ratio.anova <- aov(RATIO ~ SEX, data = mydata)
summary(ratio.anova)
ratio.tukey <- TukeyHSD(ratio.anova)
ratio.tukey$SEX
plot(ratio.tukey)
ggplot(mydata, aes(x = VOLUME, y = RATIO)) + geom_point(aes(color = SEX))
ggplot(mydata, aes(x = VOLUME, y = RATIO)) + geom_point(aes(color = CLASS))
cor(mydata$RATIO,mydata$VOLUME)
cor.test(mydata$RATIO,mydata$VOLUME)
ggplot(mydata, aes(x = VOLUME, y = SHUCK)) + geom_point()
L_SHUCK = log(mydata$SHUCK)
L_VOLUME = log(mydata$VOLUME)
cor(L_SHUCK, L_VOLUME)
ggplot(mydata, aes(x = L_VOLUME, y = L_SHUCK)) + geom_point()
lm(L_SHUCK ~ L_VOLUME)
lm(formula = L_SHUCK ~ L_VOLUME)
ggplot(mydata, aes(x=L_SHUCK, y=L_VOLUME)) + geom_point() + geom_smooth(method="lm")
ggplot(mydata, aes(x=SHUCK, y=VOLUME)) + geom_point() + geom_smooth(method="lm")
